#Program 1 - Multiplicative Inverse
# LOAD ALL SOURCE DATA
SEI 9
LW R0, RIM	# LSB DIVISOR
SEI 8
LW R1, RIM	# MSB DIVISOR
SEI 0
ADD R2, RIM	# LSB FRACTIONAL RESULT
ADD R3, RIM	# MSB FRACTIONAL RESULT
ADD R4, RIM	# COMPARE REGISTER 1 LSB
ADD R5, RIM # COMPARE REGISTER 2 MSB
SEI 1
ADD R4, RIM

# SET COUNTER
SEI 0
ADD R6, RIM

# MAIN PROCESS
FracLoop:
# -- SHIFTING COMPARE
# Comparing MSB
CLT R1, R5 
B1 ResultLarger
CEQ R1, R5
B0 ResultSmaller

# Comparing LSB
CLT R0, R4
B1 ResultLarger
CEQ R0, R4
B0 ResultSmaller

ResultLarger:
SUB R4, R0
SUB R5, R1

ResultSmaller:
SHIFTL.F R2
SHIFTL.O R3

# SHIFT COMPARE REGISTER
SHIFTL.X R4
SHIFTL.O R5

# -- ADD COUNTER
SEI 1
ADD R6, RIM
SEI 8
CEQ R6, RIM
B0 FracLoop

# STORE FRACTIONAL PART RESULT
SEI 11
SW R2, RIM
SEI 10
SW R3, RIM
